Updated Seeding Logic async function seedCandles(pair) { liveCandles[pair] = []; // Initialize empty const success = await fetchRealCandles(pair); // If real fetch failed or returned nothing, build a fake history so chart works if (!success || liveCandles[pair].length < 10) { console.log(`Simulating history for ${pair}`); let base = 1.10; // Default if (pair.includes('BTC')) base = 60000; let t = Date.now() - 100 * 60000; for(let i=0; i<100; i++) { let o = base + (Math.random()-0.5) * (base * 0.001); let c = o + (Math.random()-0.5) * (base * 0.001); liveCandles[pair].push([t + i*60000, o, Math.max(o,c), Math.min(o,c), c, 100]); } } } // 2.
